server {
        root /var/www/html;
        index index.php index.html index.htm index.nginx-debian.html;
        server_name mareksautter.com www.mareksautter.com;

        location / {
                try_files $uri $uri/ =404;
        }

        location ~ \.php$ {
                include snippets/fastcgi-php.conf;
                fastcgi_pass unix:/var/run/php/php7.2-fpm.sock;
        }

        location ~ /\.ht {
                deny all;
        }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/mareksautter.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/mareksautter.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot


}
server {
    if ($host = www.mareksautter.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    if ($host = mareksautter.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


        listen 80;
        server_name mareksautter.com www.mareksautter.com;
    return 404; # managed by Certbot




}
